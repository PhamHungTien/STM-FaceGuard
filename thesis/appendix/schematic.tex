\section*{A.2 \quad Định dạng và Timing của thông điệp UART}

Giao tiếp giữa STM32 và ESP32-S3 sử dụng protocol UART tới giản. Mỗi thông điệp kết thúc bằng ký tự newline (\texttt{\textbackslash n}), cho phép phương pháp Idle Line Detection hoạt động hiệu quả.

\begin{figure}[H]
  \centering
  \resizebox{0.92\textwidth}{!}{%
  \begin{tikzpicture}[font=\small, >=Stealth]
    %% Baseline
    \draw[thick] (0,0) -- (11,0);
    \node[left, font=\small\ttfamily] at (0,0) {TXD};

    %% Idle (before)
    \draw[thick] (0,0.4) -- (0.8,0.4);
    \node[above, font=\scriptsize] at (0.4,0.4) {Idle (HIGH)};

    %% Start bit (LOW)
    \draw[thick] (0.8,0.4) -- (0.8,0) -- (1.6,0);
    \node[below, font=\scriptsize] at (1.2,-0.05) {START};

    %% Data bits D0–D7 (01001111 = 'O' = 0x4F, LSB first)
    \foreach \idx/\bit/\lbl in {
        0/1/D0, 1/1/D1, 2/1/D2, 3/1/D3,
        4/0/D4, 5/0/D5, 6/1/D6, 7/0/D7} {
      \pgfmathsetmacro{\x}{1.6 + \idx*0.9}
      \pgfmathsetmacro{\xn}{1.6 + (\idx+1)*0.9}
      \pgfmathsetmacro{\h}{0.4*\bit}
      \draw[thick] (\x,\h) -- (\xn,\h);
      \node[above, font=\tiny] at ({(\x+\xn)/2}, \h) {\bit};
    }
    %% Transitions between data bits
    \foreach \idx/\cur/\nxt in {
        0/1/1, 1/1/1, 2/1/1, 3/1/0, 4/0/0, 5/0/1, 6/1/0} {
      \pgfmathsetmacro{\x}{1.6 + (\idx+1)*0.9}
      \pgfmathsetmacro{\hc}{0.4*\cur}
      \pgfmathsetmacro{\hn}{0.4*\nxt}
      \draw[thick] (\x,\hc) -- (\x,\hn);
    }

    %% Stop bit (HIGH)
    \pgfmathsetmacro{\stopx}{1.6+8*0.9}  % = 8.8
    \draw[thick] (\stopx,0) -- (\stopx,0.4) -- ({(\stopx+0.9)},0.4);
    \node[below, font=\scriptsize] at ({(\stopx+0.45)},-0.05) {STOP};

    %% Idle after
    \pgfmathsetmacro{\idlex}{\stopx+0.9}  % = 9.7
    \draw[thick] (\idlex,0.4) -- (11,0.4);
    \draw[thick, dashed, gray] (\idlex,0) -- (11,0);
    \node[above, font=\scriptsize, align=left] at ({(\idlex+11)/2}, 0.4)
          {Idle → Idle Line\\Detection kích hoạt};

    %% Brace annotation for data byte
    \draw[<->, thin] (1.6,-0.7) -- ({1.6+8*0.9},-0.7)
          node[midway, below, font=\scriptsize] {8 data bits};
    \draw[<->, thin] (0.8,-1.1) -- ({1.6+8*0.9+0.9},-1.1)
          node[midway, below, font=\scriptsize]
          {1 frame = 10 bit-times @ 115\,200~baud $\approx$ 87~µs/byte};
  \end{tikzpicture}}
  \caption{UART Bit Timing Diagram (ví dụ byte \texttt{0x4F} = ký tự `O'). Idle Line Detection kích hoạt khi đường truyền giữ mức HIGH liên tục sau bit STOP, đánh dấu kết thúc một gói tin hoàn chỉnh.}
  \label{fig:uart_timing}
\end{figure}

\section*{A.3 \quad Biểu đồ Timing chuyên sâu: Unlock Sequence}

Sơ đồ timing sau mô tả toàn bộ quá trình từ lúc camera phát hiện khuôn mặt đến khi relay mở khóa.

\begin{figure}[H]
  \centering
  \begin{tikzpicture}[scale=0.9, font=\footnotesize]
    % Timeline
    \draw[very thick] (0,0) -- (14,0);
    \foreach \t in {0,2,4,6,8,10,12,14} {
      \draw (\t,0.1) -- (\t,-0.1);
      \node at (\t, -0.4) {\t ms};
    }
    \node at (7, -0.8) {Time →};
    
    % Camera detection
    \draw[blue, ultra thick] (0, 1) -- (1, 1) -- (1, 1.5) -- (1.5, 1.5) -- (1.5, 1);
    \node[blue, left] at (-0.5, 1.25) {Camera};
    \node[blue] at (0.75, 1.8) {\small Frame 1 (detect face)};
    
    % ML inference (80-120 ms)
    \draw[green, ultra thick] (1, 2) -- (3, 2);
    \draw[green, dashed] (1, 1.8) -- (1, 2.2);
    \node[green, left] at (-0.5, 2) {Inference};
    \node[green] at (2, 2.3) {\small 80--120 ms};
    
    % UART transmission (5-20 ms)
    \draw[red, ultra thick] (3, 3) -- (3.5, 3);
    \draw[red, dashed] (3, 2.8) -- (3, 3.2);
    \node[red, left] at (-0.5, 3) {UART Tx};
    \node[red] at (3.25, 3.3) {\small 5--20 ms};
    
    % STM32 DMA reception (0-5 ms)
    \draw[purple, ultra thick] (3.5, 4) -- (3.7, 4);
    \draw[purple, dashed] (3.5, 3.8) -- (3.5, 4.2);
    \node[purple, left] at (-0.5, 4) {DMA RX};
    
    % FSM processing (1-5 ms)
    \draw[orange, ultra thick] (3.7, 5) -- (4, 5);
    \node[orange] at (3.85, 5.3) {\small <5 ms};
    \node[orange, left] at (-0.5, 5) {FSM};
    
    % Relay activation (10 ms electrical)
    \draw[red!80!black, ultra thick] (4, 6) -- (4.3, 6);
    \node[red!80!black, left] at (-0.5, 6) {Relay};
    \node[red!80!black] at (4.15, 6.3) {\small 10 ms};
    
    % Total: 127-210 ms
    \draw[thick, dotted] (0, -1.5) -- (4.3, -1.5);
    \node at (2.15, -2) {\bf Total Latency: 127--210 ms};
    
  \end{tikzpicture}
  \caption{End-to-End Latency Diagram: Từ khi camera phát hiện khuôn mặt đến khi relay mở. Các thành phần độc lập không chặn nhau.}
  \label{fig:unlock_timing}
\end{figure}
