\section*{A.2 \quad Định dạng và Timing của thông điệp UART}

Giao tiếp giữa STM32 và ESP32-S3 sử dụng protocol UART tới giản. Mỗi thông điệp kết thúc bằng ký tự newline (\texttt{\textbackslash n}), cho phép phương pháp Idle Line Detection hoạt động hiệu quả.

\begin{figure}[H]
  \centering
  \begin{tikzpicture}[scale=1.0, font=\small]
    % UART frame timing
    \draw[thick] (0,0) -- (10,0);
    \node at (-0.5, 0) [left] {TXD};
    
    % Idle state
    \draw[thick, dashed] (0,0.3) -- (0.5,0.3);
    \node at (0.25, 0.6) {\tiny Idle};
    
    % Start bit
    \draw[thick] (0.5,-0.3) -- (0.5,0) -- (1,-0.3);
    \node at (0.75, -0.6) {\tiny START};
    
    % Data bits (example: 'O' = 0x4F = 01001111)
    \foreach \i/\bit in {1/0, 1.5/1, 2/0, 2.5/0, 3/1, 3.5/1, 4/1, 4.5/1} {
        \draw[thick] (\i,{0.3*(\bit-0.5)}) -- (\i+0.5,{0.3*(\bit-0.5)});
        \node at (\i+0.25, {0.3*(\bit-0.5)+0.4}) {\tiny \bit};
    }
    \node at (2.75, -0.8) {Data[8]};
    
    % Stop bit
    \draw[thick] (5,-0.3) -- (5,0) -- (5.5,-0.3);
    \node at (5.25, -0.6) {\tiny STOP};
    
    % Idle again
    \draw[thick, dashed] (5.5,0.3) -- (10,0.3);
    \node at (7.75, 0.6) {\tiny Idle (Idle Line Detection fires here)};
    
    % Timing annotations
    \draw[<->] (0.5, -1.2) -- (5.5, -1.2);
    \node at (3, -1.5) {~10 bit-times @ 115.2k = ~87 µs/byte};
  \end{tikzpicture}
  \caption{UART Bit Timing Diagram. Idle line được phát hiện khi đường không có chuyển đổi bit trong khoảng thời gian cụ thể.}
  \label{fig:uart_timing}
\end{figure}

\section*{A.3 \quad Biểu đồ Timing chuyên sâu: Unlock Sequence}

Sơ đồ timing sau mô tả toàn bộ quá trình từ lúc camera phát hiện khuôn mặt đến khi relay mở khóa.

\begin{figure}[H]
  \centering
  \begin{tikzpicture}[scale=0.9, font=\footnotesize]
    % Timeline
    \draw[very thick] (0,0) -- (14,0);
    \foreach \t in {0,2,4,6,8,10,12,14} {
      \draw (\t,0.1) -- (\t,-0.1);
      \node at (\t, -0.4) {\t ms};
    }
    \node at (7, -0.8) {Time →};
    
    % Camera detection
    \draw[blue, ultra thick] (0, 1) -- (1, 1) -- (1, 1.5) -- (1.5, 1.5) -- (1.5, 1);
    \node[blue, left] at (-0.5, 1.25) {Camera};
    \node[blue] at (0.75, 1.8) {\small Frame 1 (detect face)};
    
    % ML inference (80-120 ms)
    \draw[green, ultra thick] (1, 2) -- (3, 2);
    \draw[green, dashed] (1, 1.8) -- (1, 2.2);
    \node[green, left] at (-0.5, 2) {Inference};
    \node[green] at (2, 2.3) {\small 80--120 ms};
    
    % UART transmission (5-20 ms)
    \draw[red, ultra thick] (3, 3) -- (3.5, 3);
    \draw[red, dashed] (3, 2.8) -- (3, 3.2);
    \node[red, left] at (-0.5, 3) {UART Tx};
    \node[red] at (3.25, 3.3) {\small 5--20 ms};
    
    % STM32 DMA reception (0-5 ms)
    \draw[purple, ultra thick] (3.5, 4) -- (3.7, 4);
    \draw[purple, dashed] (3.5, 3.8) -- (3.5, 4.2);
    \node[purple, left] at (-0.5, 4) {DMA RX};
    
    % FSM processing (1-5 ms)
    \draw[orange, ultra thick] (3.7, 5) -- (4, 5);
    \node[orange] at (3.85, 5.3) {\small <5 ms};
    \node[orange, left] at (-0.5, 5) {FSM};
    
    % Relay activation (10 ms electrical)
    \draw[red!80!black, ultra thick] (4, 6) -- (4.3, 6);
    \node[red!80!black, left] at (-0.5, 6) {Relay};
    \node[red!80!black] at (4.15, 6.3) {\small 10 ms};
    
    % Total: 127-210 ms
    \draw[thick, dotted] (0, -1.5) -- (4.3, -1.5);
    \node at (2.15, -2) {\bf Total Latency: 127--210 ms};
    
  \end{tikzpicture}
  \caption{End-to-End Latency Diagram: Từ khi camera phát hiện khuôn mặt đến khi relay mở. Các thành phần độc lập không chặn nhau.}
  \label{fig:unlock_timing}
\end{figure}
